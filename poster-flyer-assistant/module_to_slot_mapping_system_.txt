module_to_slot_mapping_system:
  system_name: "module_to_slot_mapping_system"
  version: "2.1.0"
  purpose: "selected modules を、selected template に沿って review しやすい block list に落とす。"

  note:
    - "file 名は slot_mapping のままだが、この版では block mapping として扱う"

  role_in_pipeline:
    position: "between_module_selection_and_copy_generation"
    inputs:
      - "selected_modules"
      - "selected_template"
      - "poster_template_slot_schema"
      - "optional_user_feedback"
    outputs:
      - "block_list_master"
      - "review_table"
      - "unassigned_modules"
      - "mapping_summary"
      - "copy_generation_targets"

  design_principles:
    - "重い slot 構造ではなく block 単位で出力する"
    - "preview で指さしやすい場所表現を優先する"
    - "review_table は block_list_master から派生表示する"
    - "人間が追加・削除・入れ替えしやすい形で出す"
    - "入りきらない module は無理に押し込まず unassigned_modules に残す"
    - "preview に見える主要 block は、すべて block_list_master に載せる"

  place_assignment_rules:
    core_message_preferred_places:
      - "表・上中央"
      - "表・上部"

    benefit_preferred_places:
      - "表・中央"
      - "表・下部"

    audience_preferred_places:
      - "表・下部"
      - "裏・上部"

    diagram_preferred_places:
      - "裏・中央"
      - "表・中央"

    proof_preferred_places:
      - "裏・下部"
      - "表・下部"

    cta_preferred_places:
      - "裏・右下"
      - "表・右下"

  major_block_inclusion_policy:
    include_if_visible:
      - "セクション見出し"
      - "導入見出し"
      - "独立した説明 block"
      - "図"
      - "比較表"
      - "CTA"
    rule:
      - "preview 上で独立して見える主要塊は 1 block として block_list_master に持つ"
      - "review_table にも同じ block を 1 行持つ"

  block_building_policy:
    - "1 block = 1 review 行 を基本とする"
    - "No, 場所, 何が書いてあるか を最小単位とする"
    - "必要なら内部で module_id を持つが、レビュー上は簡潔さを優先する"
    - "似た役割の module は 1 block に統合してもよい"
    - "block label は人間が見て分かる言い方を優先する"
    - "review_table は block_list_master を表示用に間引いたり再入力したりせず、そのまま派生させる"

  output_schema:
    block_list_master:
      description: "紙面の master block 一覧。review_table の元データ。"
      fields:
        - "no"
        - "place"
        - "content_label"
        - "module_ids"
        - "notes"
        - "is_major_visible_block"

    review_table:
      description: "人間レビュー用の構成表。block_list_master の表示用派生。"
      fields:
        - "no"
        - "place"
        - "content_label"

    unassigned_modules:
      description: "今回は block 化しなかった module"
      fields:
        - "module_id"
        - "reason"

    mapping_summary:
      description: "block 配置方針を短く要約したもの"

    copy_generation_targets:
      description: "copy generation が参照する block 単位の対象"
      fields:
        - "no"
        - "module_ids"
        - "content_label"
